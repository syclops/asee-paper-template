% ASEE requires centered titles in 14-point boldface font, with 1.5 inch margins
% on the left and right sides.
\makeatletter
\renewcommand\maketitle
{
  \begin{center}
    \begin{adjustwidth}{0.5in}{0.5in}
      \centering
      \fontsize{14}{\baselineskip}
      \bfseries
      \@title{}
    \end{adjustwidth}
  \end{center}
}
\makeatother

% reference. Customize the relevant terms for each reference type here.

% In some templates, subsections are referred to as "Subsection x.y", while
% they may be referred to as "Section x.y" instead. The following line ensures
% that sections and subsections are always referred to with the same identifier.
\let{\subsectionautorefname}{\sectionautorefname}

% Required: titlesec
% Ensure section headings are the same font and size as the rest of the
% document, as required by ASEE.
\newcommand{\formathdr}[2][\itshape]
{
  \titleformat*{#2}{\normalsize#1\raggedright}
  \titlespacing{#2}{0pt}{\baselineskip}{0pt}
}
\formathdr[\bfseries]{\section}
\formathdr{\subsection}
\formathdr{\subsubsection}
